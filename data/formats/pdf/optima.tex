\def\optimatypeface{
  \newtypeface{optima}{OptimaNovaLTPro}{}{Italic}{Bold}{Regular}
}

\def\optimatitles{
  \optimatypeface
  \newtypestyle{ahead}{optima}{}{\the\aheadsize}{rg}{rm}{sc}{po}
  \newtypestyle{bhead}{optima}{}{\the\bheadsize}{rg}{rm}{sc}{po}
  \newtypestyle{chead}{optima}{}{\the\cheadsize}{rg}{rm}{sc}{po}
  \newtypestyle{caption}{optima}{}{\the\captionsize}{rg}{rm}{sc}{po}

  \newtypestyle{runin}{optima}{}{\the\textsize}{rg}{rm}{sc}{po}

  \gdef\booktitlefont{\ahead}
  \gdef\authornamefont{\bhead}
  \gdef\parttitlefont{\ahead}
  \gdef\chaptertitlefont{\chead}
  \gdef\pageframefont{\caption}
}
